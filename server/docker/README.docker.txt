# Create a docker user defined network
docker network create -d bridge --subnet 10.31.0.0/24 docker_network

# Build postgres docker image
docker build -t mttdb-postgres -f Dockerfile.postgres .
# Start the postgres server
docker run --name mttdb-postgres --network=docker_network --ip=10.31.0.2 -e POSTGRES_PASSWORD=mtt -d mttdb-postgres 
# Test container
PGPASSWORD="mtt" psql -a --host=10.31.0.2 --dbname=mtt --username=postgres

# Build cherrypy docker image
docker build -t mttdb-cherrypy -f Dockerfile.cherrypy . 
# Start the cherrypy server
docker run --name mttdb-cherrypy --network=docker_network --ip=10.31.0.3 -dt mttdb-cherrypy /opt/mtt/server/docker/mtt_cherrypy.sh
